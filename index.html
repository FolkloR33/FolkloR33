script Powershell:
    # Paramètres
    $adresseIP = "[IP]"
    $portUDP = [7-9]
    $adresseMAC = "[MAC]"

    # Convertir l'adresse MAC en tableau de bytes
    $adresseMACBytes = $adresseMAC -split ':' | ForEach-Object { [byte]('0x' + $_) }

    # Fonction pour envoyer le paquet WOL
    function Send-WOLMagicPacket {
        param (
            [byte[]]$MACAddress
        )

        # Créer le tableau de bytes pour le paquet WOL
        $magicPacket = @(0xFF, 0xFF, 0xFF, 0xFF, 0xFF, 0xFF)

        # Répéter l'adresse MAC 16 fois dans le paquet
        1..16 | ForEach-Object {
            $magicPacket += $MACAddress
        }

        # Créer l'objet UDP client
        $udpClient = New-Object System.Net.Sockets.UdpClient

        try {
            # Envoyer le paquet WOL
            $udpClient.Connect($adresseIP, $portUDP)
            $udpClient.Send($magicPacket, $magicPacket.Length)
            Write-Host ("Paquet WOL envoyé avec succès à ${adresseIP}:${portUDP} pour ${adresseMAC}.")
        } catch {
            Write-Host ("Erreur lors de l'envoi du paquet WOL : $($_.Exception.Message)")
        } finally {
            $udpClient.Close()
        }
    }

    # Appeler la fonction Send-WOLMagicPacket avec l'adresse MAC spécifiée
    Send-WOLMagicPacket -MACAddress $adresseMACBytes
